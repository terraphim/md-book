11#[cfg(feature = "server")]
12pub async fn serve_book(
13    output_dir: String,
14    port: u16,
15    reload_tx: broadcast::Sender<()>,
16) -> Result<()> {
17    let static_files =
18        warp::fs::dir(output_dir.clone()).or(warp::fs::file(format!("{}/index.html", output_dir)));
19
20    // Add WebSocket route for live reload
21    let reload = warp::path("live-reload")
22        .and(warp::ws())
23        .map(move |ws: warp::ws::Ws| {
24            let reload_tx = reload_tx.clone();
25            ws.on_upgrade(move |socket| handle_live_reload(socket, reload_tx))
26        });
27
28    println!("Serving book at http://localhost:{}", port);
29    warp::serve(static_files.or(reload))
30        .run(([127, 0, 0, 1], port))
31        .await;
32    Ok(())
33}
34
35#[cfg(feature = "server")]
36async fn handle_live_reload(ws: WebSocket, reload_tx: broadcast::Sender<()>) {
37    let mut rx = reload_tx.subscribe();
38    let (mut ws_tx, _) = ws.split();
39
40    while rx.recv().await.is_ok() {
41        if let Err(e) = ws_tx.send(Message::text("reload")).await {
42            eprintln!("WebSocket send error: {}", e);
43            break;
44        }
45    }
46}
