152/// Load configuration from file or use defaults
153///
154/// # Errors
155///
156/// Returns an error if the configuration file cannot be read or parsed
157pub fn load_config(config_path: Option<&str>) -> anyhow::Result<BookConfig> {
158    let mut layers = vec![Layer::Env(Some("MDBOOK_".to_string()))];
159
160    // Add default book.toml if it exists
161    if std::path::Path::new("book.toml").exists() {
162        layers.push(Layer::Toml("book.toml".into()));
163    }
164
165    // Add custom config file if provided
166    if let Some(path) = config_path {
167        if std::path::Path::new(path).exists() {
168            // and is TOML
169            if std::path::Path::new(path)
170                .extension()
171                .is_some_and(|ext| ext.eq_ignore_ascii_case("toml"))
172            {
173                layers.push(Layer::Toml(path.into()));
174            } else if std::path::Path::new(path)
175                .extension()
176                .is_some_and(|ext| ext.eq_ignore_ascii_case("json"))
177            {
178                layers.push(Layer::Json(path.into()));
179            } else {
180                anyhow::bail!("Unsupported config file type: {}", path);
181            }
182        }
183    }
184
185    let config = BookConfig::with_layers(&layers)?;
186    Ok(config)
187}
